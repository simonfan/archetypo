/**
 * @module archetypo
 * @submodule view
 */

define(function (require, exports, module) {
	'use strict';

	var _ = require('lodash'),
		router = require('lowercase-backbone').router;

	var archRouter = module.exports = router.extend({

		initialize: function initializeArchRouter(options) {
			options = options || {};
			this.stringifiers = _.extend({}, this.stringifiers, options.stringifiers);
		},

		/**
		 * Hash on which route stringifiers are stored by name.
		 *
		 * @property stringifiers
		 * @type Object
		 */
		stringifiers: {
			'default': function defaultStringifier(data) {
				return '/' + JSON.stringify(data);
			}
		},

		/**
		 * Defines a stringifier method.
		 *
		 * @method stringifier
		 */
		stringifier: function defineStringifier(type, stringifier) {
			this.stringifiers[type] = stringifier;

			return this;
		},

		/**
		 * Returns a route string by using a pre-defined strngifier.
		 *
		 * @method stringify
		 * @param type String Identifier string for the stringifier.
		 * @param data
		 */
		stringify: function stringifyRoute(type, data) {

			var stringifier = this.stringifiers[type] || this.stringifiers['default'];

			return stringifier(data);
		},

		/**
		 * Intercepts the original navigate method,
		 * so that the router automatically takes advantage
		 * of stringifier methods.
		 *
		 * @method navigate
		 * @param stringifier {String}
		 * @param data {Object}
		 * @param [options] {Object}
		 */
		navigate: function navigate(stringifier, data, options) {
			var route = this.stringify(stringifier, data);

			return router.prototype.navigate.call(this, route, options);
		},
	});
});
